 77 /**
 78  * add another point to an existing pointGroup
 79  * @param {Point} point the point to add
 80  */
 81 PointGroup.prototype.add = function(point){
 82 	var c;
 83 
 84 	this.points[this.points.length] = point;
 85 	this.sort();
 86 	if (this.points[0].route.phototopo.loading){ 
 87 		return;
 88 	}
 89 	for (c=0; c < this.points.length; c++){
 90 		this.points[c].updateLabelPosition();
 91 	} 
 92 };
 93 PointGroup.prototype.sort = function(){
 94 	this.points.sort(function(a,b){
 95 		return a.route.order > b.route.order ? 1 : -1;
 96 	});

 98 }

103  * points have a natural order which is typically the order the routes are shown in the guide (eg left to right)

106 	var c,

113 			return c;

118 /*
119  * return the amount the curve should be offset when multiple curves overlap
120  */
121 PointGroup.prototype.getSplitOffset = function(point){
122 	var c, ret;
123 
124 	for(c=0; c < this.points.length; c++){
125 		if (this.points[c] === point){
126 			ret = (1 - this.points.length) / 2 + c;
127 			return ret;
128 		}
129 	}
130 	return 0;

171 /*
172  * returns the x/y coords of the cubic bezier curve that should come out of this point
173  * just make them negative to get the handle for the opposite end of the cubic curve
174  * a simplistic algortith just averages them all, a more complex one takes into account which routes actually merge or don't
175  */
176 PointGroup.prototype.getAngle = function(point){
177 
178 	// for each point get the diff of it and the next point and the previous point and add them all together
179 	// then find that angle and scale the point to the shortest path segment length
180 	
181 	var dx = 0,
182 		dy = 0,
183 		ddx = 0,
184 		ddy = 0,
185 		p,c,
186 		sqr,
187 		minSqr = 1000000,
188 		angle,
189 		dist;
190 	
191 	for(c=0; c < this.points.length; c++){
192 		p = this.points[c];
193 		if (p.nextPoint){
194 			dx = (p.nextPoint.x - p.x);
195 			dy = (p.nextPoint.y - p.y);
196 			sqr = dx*dx + dy*dy;
197 			if (sqr < minSqr){ minSqr = sqr; }
198 			ddx += dx;
199 			ddy += dy;
200 		}
201 		if (p.prevPoint){
202 			dx = (p.prevPoint.x - p.x);
203 			dy = (p.prevPoint.y - p.y);
204 			sqr = dx*dx + dy*dy;
205 			if (sqr < minSqr){ minSqr = sqr; }
206 			ddx -= dx;
207 			ddy -= dy;
208 		}
209 	}
210 	
211 	angle = Math.atan2(ddx, ddy);
212 	dist = Math.sqrt(minSqr) * 0.4;
213 	
214 	ddx = dist * Math.sin(angle);
215 	ddy = dist * Math.cos(angle);
216 		
217 	return {dx:ddx,dy:ddy};
218 };

251 	this.pointGroup = this.route.phototopo.getPointGroup(this);
252 	
253 	
254 	styles = this.route.phototopo.styles;
255 
256 	if (this.route.phototopo.options.editable){
257 
258

280 			this.point.select();
281 	
282 		}

285 			if (selectedRoute && selectedRoute !== this.point.route){
286 				return;

289 			circle.attr({cx: pos.x, cy: pos.y});
290 		}

292 			var selectedRoute = this.point.route.phototopo.selectedRoute;
293 		  	$(this.point.route.phototopo.photoEl).removeClass('dragging');

296 				return;
297 			}
298 			this.point.setStyle();
299 		}
300 		
301 		circle.drag(dragMove, dragStart, dragEnd); 
302 		circle.mouseover(function(){
303 			// is it the last point of a routes?
304 			if (this.point.nextPoint){
305 				$('#phototopoContextMenu .hidden').removeClass('disabled');
306 				$('#phototopoContextMenu .jumpoff').addClass('disabled');
307 			} else {
308 				$('#phototopoContextMenu .hidden').addClass('disabled');
309 				$('#phototopoContextMenu .jumpoff').removeClass('disabled');
310 			}
311 
312 			$(this.point.route.phototopo.photoEl).addClass('point');
313 
314 			//this.point.setStyle();
315 			this.point.route.onmouseover(this.point);
316 			this.animate(styles.handleHover, 100);
317 		});
318 		circle.mouseout(function(){
319 		  	$(this.point.route.phototopo.photoEl).removeClass('point');
320 			this.point.route.onmouseout(this.point);
321 			this.point.setStyle();
322 		});
323 		circle.click(function(e){
324 			var route = this.point.route.phototopo.selectedRoute;
325 			// don't allow adding a point in the current route back to itself
326 			if (this.point.route === route){
327 				return;
328 			}
329 			if (route){
330 				route.addAfter(this.point.route.phototopo.selectedPoint, this.point.x, this.point.y);
331 			} else {
332 				this.point.select();
333 			}
334 		});
335 		circle.dblclick(function(){
336 			this.point.remove();
337 		});
338 	
339 	
340 		$(circle.node).contextMenu({
341 			menu: 'phototopoContextMenu'
342 		},
343 			function(action, el, pos) {
344 				point.setType(action);
345 				point.pointGroup.redraw();
346 				point.route.phototopo.saveData();
347 
348 		});
349 		
350 		
351 	}
352 	this.setType(this.type);
353 }

443 /**
444  * @private
445  * remove a point from its route
446  */
447 Point.prototype.remove = function(){
448 
449 	var p = this,
450 		path = null,
451 		r, c;
452 	p.remove = 'todo';
453 
454 	// remove handle from raphael
455 	p.circle.remove();
456 
457 	// remove from point group
458 	p.pointGroup.remove(p);
459 	p.pointGroup.redraw();
460 	p.pointGroup = null;
461 
462 	// remove all handlers for the point and refs to/from dom
463 	r = this.route;
464 	
465 	// remove point from array
466 	r.points.splice(p.position, 1);
467 
468 	// fix all position's of points after this one
469 	for(c=p.position; c<r.points.length; c++){
470 		r.points[c].position = c;
471 	}
472 
473 	// if one path then remove and relink
474 	if (p.prevPath){
475 		// remove prev path and relink
476 		path = p.prevPath;
477 		path.remove = 'now!';
478 
479 		// fix point refs
480 		p.prevPoint.nextPoint = p.nextPoint;
481 		p.prevPoint.nextPath = p.nextPath;
482 
483 		// fix path points
484 		if (p.nextPoint){	
485 			p.nextPoint.prevPoint = p.prevPoint;
486 		}
487 		if (p.nextPath){	
488 			p.nextPath.point1 = p.prevPoint;
489 		}
490 
491 		r.paths.splice(p.position-1, 1);
492 		path.curve.remove();
493 		path.outline.remove();
494 		path.ghost.remove();
495 
496 		// select prev point
497 		p.prevPoint.select();
498 	} else if (p.nextPath){
499 		// it is the first
500 		// if the first point then move the label to the next point
501 		// select next point
502 		path = p.nextPath;
503 		path.remove = 'now2!';
504 		
505 		p.nextPoint.prevPoint = null;
506 		p.nextPoint.prevPath = null;
507 		
508 		r.paths.splice(p.position, 1);
509 		path.curve.remove();
510 		path.outline.remove();
511 		path.ghost.remove();
512 
513 		// select next point
514 		p.nextPoint.select();
515 	
516 	} else {
517 		
518 		// just one point so delete it
519 	}
520 	if (this.labelEl){
521 		this.labelEl.parentNode.removeChild(this.labelEl);
522 		this.labelEl = null;
523 	}
524 	if (this.iconEl){
525 		this.iconEl.parentNode.removeChild(this.iconEl);
526 		this.iconEl = null;
527 	}
528 
529 	if (this.route.phototopo.selectedPoint === this){
530 		this.route.phototopo.selectedPoint = null;
531 	}
532 
533 	p.route.redraw();
534 	if (p.nextPoint){
535 		p.nextPoint.pointGroup.redraw();
536 	}
537 	if (p.prevPoint){
538 		p.prevPoint.pointGroup.redraw();
539 	}
540 	this.route.phototopo.saveData();
541 	$(this.route.phototopo.photoEl).removeClass('point');
542 	this.route.phototopo.updateCursor();
543 
544 };

546 /**
547  * @private
548  * updates the labels position
549  */
550 Point.prototype.updateLabelPosition = function(){
551 	
552 	var div = this.labelEl,
553 		offsetX, offsetY,
554 		width, top, left,
555 		maxTop, maxLeft,
556 		phototopo = this.route.phototopo;
557 	if (!div){ return; }
558 	
559 	offsetX = this.pointGroup.getSplitOffset(this) * div.offsetWidth;
560 	width = (div.offsetWidth) / 2;
561 	offsetY = phototopo.options.thickness;	
562 
563 	left = this.x - width + offsetX;
564 	top  = this.y + offsetY;
565 	
566 	maxTop = phototopo.shownHeight - div.offsetHeight;
567 	if (top>maxTop){ top = this.y - offsetY - div.offsetHeight; }
568 	
569 	div.style.left = left + 'px';
570 	div.style.top = top + 'px';
571 };

628 /**
629  * @private
630  * attempts to move the Point to a new location - it may not move due to 'stickyness' to itself and other points
631  */
632 Point.prototype.moveTo = function(x,y){
633 
634 	if (this.x === x && this.y === y){
635 		return { x: x, y: y };
636 	}
637 	if (x === NaN || y === NaN){
638 		return { x: this.x, y: this.y };
639 	}
640 
641 	this.pointGroup.remove(this);
642 	this.pointGroup.redraw();
643 
644 	this.x = x;
645 	this.y = y;
646 
647 	this.pointGroup = this.route.phototopo.getPointGroup(this);
648 	
649 	// retrive the x and y from the point group which might not be what we specified
650 	this.x = this.pointGroup.x;
651 	this.y = this.pointGroup.y;
652 
653 
654 	this.pointGroup.redraw();
655 	if (this.nextPoint){
656 		this.nextPoint.pointGroup.redraw();
657 	}
658 	if (this.prevPoint){
659 		this.prevPoint.pointGroup.redraw();
660 	}
661 
662 	if (this.labelEl){
663 		this.updateLabelPosition();
664 	}
665 
666 	this.updateIconPosition();
667 
668 	this.route.phototopo.saveData();
669 
670 	return { x: this.x, y: this.y };
671 
672 
673 };

682 /**
683  * @private
684  * a path connects two points
685  * @constructor 
686  * @param {Point} point1 The starting point
687  * @param {Point} point2 The ending point
688  */
689 function Path(point1, point2){
690 
691 	function getID(){
692 		var id=0;
693 		return id++;
694 	}
695 
696 
697 
698 
699 	var route, offset, path, phototopo;
700 
701 	this.point1 = point1;
702 	this.point2 = point2;
703 	this.id = getID();
704 	this.svg_part = '';
705 	this.point1.nextPath = this;
706 	this.point2.prevPath = this;
707 
708 	phototopo = this.point1.route.phototopo;	
709 
710 
711 	path = 'M'+this.point1.x+' '+this.point1.y+' L'+this.point2.x+' '+this.point2.y;
712 	this.svg_part = path;
713 
714 	this.pathPart = ' L'+this.point2.x+' '+this.point2.y;
715 
716 
717 	this.outline = phototopo.canvas.path(path);
718 	this.ghost   = phototopo.canvas.path(path);
719 	
720 	this.outline.toBack();
721 	this.ghost.toBack();
722 	if (this.point1.route.phototopo.bg){
723 		this.point1.route.phototopo.bg.toBack();
724 	}
725 	this.point1.route.phototopo.fill.toBack();
726 
727 
728 	this.curve      = phototopo.canvas.path(path);
729 	this.outline.attr(phototopo.styles.outline);
730 	if (this.point1.route.autoColorBorder){
731 		this.outline.attr('stroke', this.point1.route.autoColorBorder);
732 	}
733 	
734 	
735 	this.ghost.attr  (phototopo.styles.ghost);

787 /*
788  * @private
789  * changes the start point
790  */
791 Path.prototype.redraw = function(point){
792 	var handle1 = this.point1.pointGroup.getAngle(this.point1),
793 		handle2 = this.point2.pointGroup.getAngle(this.point2),
794 		points,
795 		path,
796 		phototopo = this.point1.route.phototopo,
797 		path_finish = '',
798 		off1, off2, thickness,
799 		ddx, ddy,
800 		delta, angle, aWidth, aHeight, size,
801 		ex, ey;
802 
803 	points = [
804 		this.point1,
805 		{x: this.point1.x + handle1.dx, y:this.point1.y + handle1.dy},
806 		{x: this.point2.x - handle2.dx, y:this.point2.y - handle2.dy},
807 		this.point2
808 	];
809 	
810 	
811 	/*
812 	 * takes a set of points that defines a bezier curve and offsets it
813 	 */
814 	function getBezierOffset(points, offset1, offset2){
815 	
816 		function secant(theta){
817 			return 1 / Math.cos(theta);
818 		}
819 	
820 		var res = [{}],
821 			c,
822 			angles = [],
823 			size = points.length -1,
824 			offset,
825 			offSec,
826 			angleAvg;
827 		for(c=0; c<3; c++){
828 			angles[c] = Math.atan2(points[c+1].y - points[c].y, points[c+1].x - points[c].x);
829 		}
830 		for(c=1; c<size; c++){
831 			offset = (offset1 * (size-c)) / size + (offset2 * c)/size;
832 			offSec = offset * secant((angles[c] - angles[c-1])/2);
833 			angleAvg = (angles[c]+angles[c-1])/2;
834 			res[c] = {
835 				x: points[c].x - offSec * Math.sin(angleAvg),
836 				y: points[c].y + offSec * Math.cos(angleAvg)
837 			};
838 		}
839 		res[0] = {
840 			x: points[0].x - offset1 * Math.sin(angles[0]),
841 			y: points[0].y + offset1 * Math.cos(angles[0])
842 		};
843 		res[size] = {
844 			x: points[size].x - offset2 * Math.sin(angles[size-1]),
845 			y: points[size].y + offset2 * Math.cos(angles[size-1])
846 		};
847 		for(c=0; c<res.length; c++){
848 			res[c].x = Math.round(res[c].x);
849 			res[c].y = Math.round(res[c].y);
850 		}
851 		return res;
852 	}
853 
854 
855 
856 	if (phototopo.options.seperateRoutes){
857 		thickness = phototopo.options.thickness;
858 		off1 = this.point1.pointGroup.getSplitOffset(this.point1) * thickness * 1.4;
859 		off2 = this.point2.pointGroup.getSplitOffset(this.point2) * thickness * 1.4;
860 		points = getBezierOffset(points, off1, off2);
861 	}
862 
863 	this.svg_part = 
864 		'C' + points[1].x + ' '+points[1].y +
865 		' ' + points[2].x + ' '+points[2].y +
866 		' ' + points[3].x + ' '+points[3].y;
867 		
868 	path = 'M' + points[0].x + ' '+points[0].y + this.svg_part;
869 		
870 
871 	function offset(angle, x, y, dx, dy){
872 		ddx = Math.round((x - Math.sin(angle)*dx - Math.cos(angle)*dy)*10)/10;
873 		ddy = Math.round((y + Math.cos(angle)*dx - Math.sin(angle)*dy)*10)/10;
874 		return 'L'+ddx+' '+ddy+' ';
875 	}
876 
877 	delta = this.point2.pointGroup.getAngle();
878 	angle = Math.atan2(delta.dy, delta.dx);
879 	size = phototopo.options.thickness * 0.5;
880 
881 	// x,y of end point
882 	ex = points[3].x;
883 	ey = points[3].y;
884 	// If this is the end of the Path then draw an arrow head
885 	if (!this.point2.nextPoint && this.point2.type === 'jumpoff'){
886 		aWidth = size*4;
887 		aHeight = size*0.1;
888 
889 		path_finish += offset(angle, ex, ey,       0,  -aHeight   ); // bottom middle
890 		path_finish += offset(angle, ex, ey, -aWidth,  -aHeight   ); // bottom left
891 		path_finish += offset(angle, ex, ey, -aWidth,   aHeight   ); // top left
892 		path_finish += offset(angle, ex, ey,  aWidth,   aHeight   ); // top right
893 		path_finish += offset(angle, ex, ey,  aWidth,  -aHeight   ); // bottom left
894 		path_finish += offset(angle, ex, ey, -aWidth,  -aHeight   ); // bottom left
895 	
896 	// draw a T bar stop
897 	} else if (!this.point2.nextPoint){
898 		aWidth  = size*1.5;
899 		aHeight = size*1.5;
900 		path_finish += offset(angle, ex, ey,       0,  size*1.2   ); // middle
901 		path_finish += offset(angle, ex, ey, -aWidth,  aHeight    ); // bottom left
902 		path_finish += offset(angle, ex, ey,  aWidth,  aHeight    ); // bottom right
903 		path_finish += offset(angle, ex, ey,       0, -size*2.3   ); // top
904 		path_finish += offset(angle, ex, ey, -aWidth,  aHeight    ); // bottom left
905 		path_finish += offset(angle, ex, ey,  aWidth,  aHeight    ); // bottom right
906 	}
907 	this.svg_part += path_finish;
908 	path += path_finish;
909 
910 	this.curve.attr('path', path);
911 	
912 	if (this.point1.type === 'hidden'){
913 		this.curve.attr(phototopo.styles.strokeHidden);
914 	} else {
915 		this.curve.attr(phototopo.styles.strokeVisible);
916 	}
917 	
918 	this.outline.attr('path', path);
919 	this.ghost.attr('path', path);
920 };

949 /**
950  * @private
951  */
952 Route.prototype.addPoint = function(x,y,type,offset){
953 
954 	var c, p, path;
955 
956 	// if offset is not specified then add it at the end of the route
957 	if (offset === undefined || offset === null){
958 		offset = this.points.length;
959 	}
960 	x = Math.floor(x);
961 	y = Math.floor(y);
962 
963 	p = new Point(this, x, y, type, offset);
964 
965 	// fix next and prev pointers
966 	p.prevPoint = this.points[offset-1];
967 	if (p.prevPoint){
968 		p.nextPoint = p.prevPoint.nextPoint;
969 		p.nextPath = p.prevPoint.nextPath;
970 		p.prevPoint.nextPoint = p;
971 	}
972 
973 	// what about p.next??
974 	
975 	// loop through and fix positions
976 	
977 	//each point has both next and prev points AND paths
978 
979 	// add this point into the point list
980 	this.points.splice(offset, 0, p);
981 
982 
983 	// recalc the points positions
984 	for(c = this.points.length-1; c>offset; c--){
985 		this.points[c].position = c;
986 	}
987 	if (p.nextPoint){
988 		p.nextPoint.prevPoint = p;
989 	}
990 
991 	// if more than one point make a path
992 	if(this.points.length >1){
993 		path = new Path(this.points[offset-1], this.points[offset]);
994 		this.paths.splice(offset-1, 0, path);
995 		if (this.paths[offset]){
996 			this.paths[offset].point1 = p;
997 		}
998 	}
999 	
1000 	this.phototopo.saveData();
1001 	this.phototopo.updateHint();
1002 
1003 	if (p.nextPoint){
1004 		p.nextPoint.pointGroup.redraw();
1005 	}
1006 	if (p.prevPoint){
1007 		p.prevPoint.pointGroup.redraw();
1008 	}
1009 	p.pointGroup.redraw();
1010 
1011 	return p;
1012 };
